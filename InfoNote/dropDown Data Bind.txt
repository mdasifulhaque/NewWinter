
	@RequestMapping(value = "/reg_subject", method = RequestMethod.GET)
	public String viewSubjectRegForm(Model model) {
		
		SubjectInfo subjectInfo = new SubjectInfo();
		
		List<PhoneNumber> objList = new ArrayList<PhoneNumber>();
		
		for(int i=0; i<10; i++) {
			PhoneNumber obj = new PhoneNumber();
			obj.setId(10001L + i);
			obj.setPhoneNumber("01676784124" + i);
			objList.add(obj);
		}
		
		model.addAttribute("subjectInfo", subjectInfo);
		model.addAttribute("objList", objList);
		return Constants.SUBJECT_REGISTRATION_FORM;
	}
